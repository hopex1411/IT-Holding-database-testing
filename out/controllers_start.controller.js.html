/**
 * This function redirects the user to another route if a specifik condition is met. If none of the conditions are met, the function will just redirect the user to the home page '/'.
 * @param {Object} req
 * @param {Function} res
 * @param {Function} next
 */
exports.controller_module2 = function (req, res, next) {
    if (req.fields.stuff2 === 's2') {
        res.redirect('/s2');
    } else if (req.fields.stuff2 === 's3') {
        res.redirect('/s3');
    } else {
        res.redirect('/');
    }
};
